🔎 Imperative API — Tool Registrations

Tool chaining pattern
This page registers three tools that form a natural workflow: search_eras → get_era_details → compare_eras. The agent can search the catalog, drill into a specific era, and then compare multiple candidates side by side. Each tool returns structured data and also updates the visible UI so the human sees the same results.

📄 Full Tool Registrations

// ── Tool 1: search_eras ──
navigator.modelContext.registerTool({
  name: "search_eras",
  description: "Search and filter the ChronoVoyage era catalog. You can
    filter by region, danger level, clearance required, and/or free-text
    query. All filters are optional and combinable (AND logic across
    categories). Returns a list of matching eras with summary info.
    If no filters are provided, returns all eras.",
  inputSchema: {
    type: "object",
    properties: {
      query: {
        type: "string",
        description: "Free-text search query. Matches against era names,
          descriptions, highlights, and cultural notes. Examples:
          'ancient civilizations', 'peaceful', 'samurai', 'futuristic'.
          Optional — omit to skip text filtering."
      },
      region: {
        type: "string",
        enum: ["Europe", "Asia", "Americas", "Africa", "Oceania"],
        description: "Filter by geographical region. Optional."
      },
      dangerLevel: {
        type: "number",
        description: "Filter by maximum danger level (1=safest, 5=most
          dangerous). Returns eras at or below this level. Optional."
      },
      clearanceRequired: {
        type: "number",
        description: "Filter by maximum clearance level required (1-3).
          Returns eras that require this level or lower. Optional."
      }
    },
    required: []
  },
  async execute(params) {
    const results = filterEras(params);
    // Updates both the returned data and the visible UI
    return {
      content: [{ type: "text", text: JSON.stringify({
        matchCount: results.length,
        eras: results.map(era => ({
          id: era.id, name: era.name,
          timePeriod, region, dangerLevel, requiredClearance,
          briefDescription
        }))
      }) }]
    };
  }
});


// ── Tool 2: get_era_details ──
navigator.modelContext.registerTool({
  name: "get_era_details",
  description: "Get full details about a specific era. Accepts an era ID
    (e.g. 'ancient-rome') or an era name (e.g. 'Ancient Rome').
    Use search_eras first to find matching eras if you don't have
    an ID or name. Returns complete era information including
    highlights, risks, prerequisites, available packages with
    pricing, and cultural notes.",
  inputSchema: {
    type: "object",
    properties: {
      id: {
        type: "string",
        description: "The era's unique ID (e.g. 'ancient-rome',
          'edo-japan'). Use this if you have it."
      },
      name: {
        type: "string",
        description: "The era's display name or a partial match
          (e.g. 'Rome', 'Jazz Age'). Used as fallback if id
          is not provided."
      }
    },
    required: []
  },
  async execute(params) {
    const era = resolveEra(params);
    // Opens the inline detail panel in the UI
    return {
      content: [{ type: "text", text: JSON.stringify({
        id, name, timePeriod, region, dangerLevel,
        requiredClearance, briefDescription, highlights,
        risks, culturalNotes, prerequisites,
        availablePackages, applicableAddOns, priceMultiplier
      }) }]
    };
  }
});


// ── Tool 3: compare_eras ──
navigator.modelContext.registerTool({
  name: "compare_eras",
  description: "Compare 2-3 eras side by side. Accepts era IDs or names.
    Returns a structured comparison object with attributes for each
    era so you can reason about the differences and make
    recommendations. Use search_eras or get_era_details first if
    you need to identify which eras to compare.",
  inputSchema: {
    type: "object",
    properties: {
      eras: {
        type: "array",
        items: { type: "string" },
        description: "An array of 2-3 era IDs or names to compare.
          Example: ['ancient-rome', 'edo-japan'] or
          ['Rome', 'Edo Japan'].",
        minItems: 2,
        maxItems: 3
      }
    },
    required: ["eras"]
  },
  async execute(params) {
    const resolved = resolveAllEras(params.eras);
    // Populates the comparison tray and shows the table in the UI
    return {
      content: [{ type: "text", text: JSON.stringify({
        comparison: resolved.map(era => ({
          id, name, timePeriod, region, dangerLevel,
          requiredClearance, highlightCount, riskCount,
          prerequisiteCount, prerequisites, priceRange
        }))
      }) }]
    };
  }
});
